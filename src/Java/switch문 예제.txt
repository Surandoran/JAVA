import java.util.*;

public class Test1 {
	public static void main(String[] args) {
		Scanner input = new Scanner(System.in);
		
		System.out.println("숫자를 입력하세요 >> ");
		int number = input.nextInt();
		
		switch(number) {
			case 10, 70, 80, 100:
			System.out.println("number는 10이다");
				break;
			case 20:
			System.out.println("number는 20이다");
				break;
			default:
				System.out.println("10도 20도 아니다");
		}
		
		
	}

}
