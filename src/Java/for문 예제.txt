2번
System.out.println("정수 n을 입력하세요");
		int n = input.nextInt();
		int sum = 0;
		//1부터 n까지 모든숫자를 탐색한다
		for (int i = 1; i <= n; i++) {
			if (n % i == 0) {
				sum += i; //sum에 찾은 약수 i 를 더해준다
				System.out.println(sum);
			}
		}
		System.out.println(n + "의 약수의 합" + sum);
		
		
		
		
		
	}
}
